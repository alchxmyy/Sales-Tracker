let currentTarget = 100; // Temporary default, will be updated from sheet let currentSales = 0; async function loadTarget() { try { const response = await fetch(SCRIPT_URL + '?action=target'); currentTarget = parseInt(await response.text()); // Update target display document.getElementById('targetInfo').innerText = 'Monthly Target: ' + currentTarget + ' policies'; document.getElementById('targetNumber').innerText = currentTarget; // After loading target, load sales loadSales(); } catch(error) { console.error('Error loading target:', error); document.getElementById('targetInfo').innerText = 'Error loading target'; // Still try to load sales with default target loadSales(); } } async function loadSales() { try { const response = await fetch(SCRIPT_URL + '?action=get'); currentSales = parseInt(await response.text()); document.getElementById('sales').innerText = currentSales; updateProgress(); } catch(error) { document.getElementById('status').innerText = 'Error loading: ' + error.message; } } function updateProgress() { const percent = (currentSales / currentTarget) * 100; const percentDisplay = Math.min(percent, 100).toFixed(1); document.getElementById('progressPercent').innerText = percentDisplay + '%'; document.getElementById('progressBar').style.width = Math.min(percent, 100) + '%'; document.getElementById('progressBar').innerText = percentDisplay + '%'; } async function addOne() { try { const response = await fetch(SCRIPT_URL + '?action=add'); const result = await response.text(); if(result === 'TARGET_REACHED') { document.getElementById('status').innerText = '🎉 Target reached! Congratulations team! 🎉'; } else { document.getElementById('status').innerText = 'Sale recorded! +1 ✅'; loadSales(); } } catch(error) { document.getElementById('status').innerText = 'Error: ' + error.message; } } async function resetSales() { if(confirm('⚠️ Are you sure you want to reset ALL sales? This affects everyone!')) { try { await fetch(SCRIPT_URL + '?action=reset'); loadSales(); document.getElementById('status').innerText = 'Sales reset to 0'; } catch(error) { document.getElementById('status').innerText = 'Error resetting: ' + error.message; } } } // Start by loading the target from Google Sheet loadTarget();
